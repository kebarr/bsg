111  template<typename T, typename T2=T>
112  struct accumulator {
113  T2 sum; // we could plug in a more accurate type for the sum
114  T S;
115  T M;
116  size_t N;
117 
118  // default constructor initializes all values
119  accumulator() : sum(0), S(0), M(0), N(0) {}
120 
121  // add another number
122  T2 operator()(const T &x) {
123  ++N;
124  sum += x;
125  T Mprev = M;
126  M += (x - Mprev) / N;
127  S += (x - Mprev) * (x - M);
128  return sum;
129  }
130 
131  T mean() const { return sum / N; }
132 
133  T variance() const { return S / (N - 1); }
134 
135  // operator<< to print the statistics to screen:
136  // denoted friend just to be able to write this inside
137  // the class definition and thus not to need to write
138  // the template specification of accumulator...
139  friend std::ostream &operator<<(std::ostream &out, const accumulator &a) {
140  if (a.N > 0)
141  out << "N\t\t\t= " << a.N << std::endl << "sum\t\t\t= " << a.sum << std::endl << "mean\t\t= "
142  << std::fixed << std::setprecision(2) << a.mean() << std::endl;
143  if (a.N > 1)
144  out << "sd\t\t\t= " << std::fixed << std::setprecision(2) << std::sqrt(a.variance()) << std::endl;
145  else
146  out << "sd\t\t\t= " << std::fixed << std::setprecision(2) << 0 << std::endl;
147  return out;
148  }

191  std::vector<Block> getBlocks(const std::vector<Match> &matches) const {
192  std::vector<Block> blocks;
193  uint32_t blkDif(0), offDif(0);
194  if (!matches.empty()) {
195  auto prev = matches.cbegin();
196  Block blk(prev->readPos, prev->dirContig, prev->offset, 1, 0);
197  auto curr = prev;
198  ++curr;
199  for (; curr != matches.cend(); ++curr) {
200  if (blk.contigID == curr->dirContig and abs(curr->offset - prev->offset) < offset_limit) {
201  blk.count++;
202  blk.setEnd(curr->readPos);
203  } else {
204  if (blk.contigID != curr->dirContig) blkDif++;
205  if (abs(curr->offset - prev->offset) >= offset_limit) offDif++;
206  blocks.push_back(blk);
207  blk = Block(curr->readPos, curr->dirContig, curr->offset, 1, 0);
208  }
209  prev = curr;
210  }
211  blocks.push_back(blk);
212  }
213  std::sort(blocks.begin(), blocks.end(), typename Block::byReadPos());
214 // read_block_stats << blkDif << "," << offDif; // First 2 values, missing validBlocks
215  return blocks;
216  }

218  std::vector<Match> getMatches() {
219  std::vector<Match> matches;
220  int64_t offset(0);
221  uint64_t p(0);
222  while (p < currentFileRecord.seq.size()) {
223  //fkmer: grows from the right (LSB)
224  //rkmer: grows from the left (MSB)
225  fillKBuf(currentFileRecord.seq[p], p, fkmer, rkmer, last_unknown);
226  p++;
227  if (last_unknown >= K) {
228  // Generate all the kmer-contig matches with Direction, Contig, Offset and readPosition
229  std::pair<KmerIDX,int> read_kmer (
230  (fkmer < rkmer) ? std::make_pair(KmerIDX(fkmer), 1) : std::make_pair(KmerIDX(rkmer), -1));
231  auto kmer = kmers.find(read_kmer.first);
232  if (kmer != kmers.end()) { // If the kmer from the read is in the unitigs
233  if (kmer->contigID * read_kmer.second > 0) {
234  offset = kmer->pos - p;
235  } else {
236  offset = kmer->pos - currentFileRecord.seq.size() + p;
237  }
238  matches.emplace_back(kmer->contigID * read_kmer.second, offset, p);
239  }
240 // ctg_read << ((kmer != kmers.end()) ? abs(kmer->contigID) : 0) << " ";
241  }
242  }
243  // Sort matches by contig and offset
244  std::sort(matches.begin(), matches.end(), typename Match::byCtgOffset());
245  return matches;
