15 bool check_or_create_directory(std::string &output_prefix) {
16  if (output_prefix.back() != '/') {
17  output_prefix.push_back('/');
18  }
19  struct stat sb{};
20  bool validate_dir(false);
21  if (stat(output_prefix.c_str(), &sb) != 0) {
22  if (errno == ENOENT) {
23  mode_t mask = umask(0);
24  umask(mask);
25  mkdir(output_prefix.c_str(), mode_t(0777 - mask));
26  validate_dir = true;
27  }
28  if (stat(output_prefix.c_str(), &sb) != 0) {
29  perror(output_prefix.c_str());
30  validate_dir = false;
31  }
32  } else if (!S_ISDIR(sb.st_mode)) {
33  std::cout << output_prefix << " is not a directory " << std::endl;
34  } else {
35  validate_dir = true;
36  }
37  return validate_dir;
38 }
