132  std::vector<bool> active_file(files_size,true);

135  std::vector<unsigned int> count_element_from_file(files_size,0);
136  std::vector<unsigned int> size_element_from_file(files_size,0);
137  std::vector<unsigned int> seen_element_from_file(files_size,0);
138  uint64_t outCount(0), count0(0), bufferSize(10000000);
139  uint64_t outBufferSize(50000000);
140 
141  ::write(out, &outCount, sizeof(outCount));
142 
143  auto *outfreqs=(RecordType*) malloc(outBufferSize* sizeof(RecordType));
144 
145  for (auto i = 0; i < files_size; i++) {
146  in_fds[i] = ::open(files[i].c_str(), O_RDONLY);
147  uint64_t size;
148  ::read(in_fds[i], &size, sizeof(size));
149  //std::cout << files[i] << " size " << size << " in disk " << sizeof(size) + size* sizeof(RecordType) << std::endl;
150  next_element_from_file[i] = (RecordType*) malloc(bufferSize* sizeof(RecordType));
151  count_element_from_file[i] = 0;
152  size_element_from_file[i]=
153  ::read(in_fds[i], next_element_from_file[i], bufferSize* sizeof(RecordType)) / sizeof(RecordType);
154  active_file[i] = true;
155  }
156 
157  RecordType min_element;
158  unsigned int min = 0;
159  for (unsigned int i = 1; i < files_size; ++i) {
160  if (next_element_from_file[i] < next_element_from_file[i-1]) min = i;
161  }
162 
163  RecordType current_element;
164  if (elements[memoryElement] < next_element_from_file[min][0]) {
165  current_element = elements[memoryElement];
166  } else {
167  current_element = next_element_from_file[min][count_element_from_file[min]];
168  count_element_from_file[min];
169  }
170  current_element.count=0;
171 
172 
173  bool active;
174  do {
175  active=false;
176  min_element = RecordType();
177 
178  for (int i=0; i<files_size; ++i) {
179  if (size_element_from_file[i] <= count_element_from_file[i]) continue;
180 
181  if (current_element == next_element_from_file[i][count_element_from_file[i]]) {
182  current_element.merge(next_element_from_file[i][count_element_from_file[i]]);
183  count_element_from_file[i]++;
184  if (count_element_from_file[i] == size_element_from_file[i]) {
185  size_element_from_file[i] =
186  ::read(in_fds[i], (char *) next_element_from_file[i], bufferSize*sizeof(RecordType)) / sizeof(RecordType);
187  seen_element_from_file[i]+=count_element_from_file[i];
188  count_element_from_file[i] = 0;
189  }
190  }
191  if (count_element_from_file[i] < size_element_from_file[i]) {
192  if (!(next_element_from_file[i][count_element_from_file[i]] > min_element)){
193  active=true;
194  min_element = next_element_from_file[i][count_element_from_file[i]];
195  }
196  }
197  }
198  if (memoryElement < numMemoryElements) {
199  if (current_element == elements[memoryElement]) {
200  current_element.merge(elements[memoryElement]);
201  memoryElement++;
202  }
203  }
204  if (memoryElement<numMemoryElements and !(elements[memoryElement] > min_element)) {
205  active=true;
206  min_element = elements[memoryElement];
207  }
208 
209  outfreqs[count0] = current_element;
210  ++count0;
211  if (count0 == outBufferSize) {
212  ::write(out, (char *) outfreqs, outBufferSize * sizeof(RecordType));
213  outCount += count0;
214  count0 = 0;
215  }
216  current_element = min_element;
217  current_element.count=0;
218  } while ( active );
219 
220  if (count0 > 0) {
221  ::write(out, (char *) outfreqs, count0 * sizeof(RecordType));
222  outCount += count0;
223  }
224  lseek(out, 0, SEEK_SET);
225  ::write(out, (char *) &outCount, sizeof(outCount));
226  free(outfreqs);
227  close(out);
228 
229  for (auto i = 0; i < files_size; i++) {
230  free(next_element_from_file[i]);
231  std::remove(files[i].c_str());
232  }
233  return outCount;

434  void collapse(std::vector<RecordType> &_elements){
435  if (_elements.size() == 0) return;
436  std::sort(_elements.begin(), _elements.end());
437  typename std::vector<RecordType>::iterator writePtr;
438  writePtr = _elements.begin();
439  typename std::vector<RecordType>::const_iterator endPtr;
440  endPtr = _elements.end();
441 
442  // Keep the total in the first equal KMer
443  typename std::vector<RecordType>::const_iterator readPtr;
444  for (readPtr = _elements.begin(); readPtr != endPtr; ++writePtr) {
445  *writePtr = *readPtr;
446  ++readPtr;
447  while (readPtr != endPtr and *writePtr == *readPtr) {
448  writePtr->merge(*readPtr);
449  ++readPtr;
450  }
451  }
452  // After accumulating the values on the first KMer, remove all duplicates using unique
453  // this operation is safe because the first copy is kept for all unique values
454  // resize the object by erasing the duplicates
455  _elements.resize(std::distance(_elements.begin(), writePtr));
456  }
