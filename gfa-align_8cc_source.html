17 bool check_or_create_directory(std::string &output_prefix) {
18  if (output_prefix.back() != '/') {
19  output_prefix.push_back('/');
20  }
21  struct stat sb{};
22  bool validate_dir(false);
23  if (stat(output_prefix.c_str(), &sb) != 0) {
24  if (errno == ENOENT) {
25  mode_t mask = umask(0);
26  umask(mask);
27  mkdir(output_prefix.c_str(), mode_t(0777 - mask));
28  validate_dir = true;
29  }
30  if (stat(output_prefix.c_str(), &sb) != 0) {
31  perror(output_prefix.c_str());
32  validate_dir = false;
33  }
34  } else if (!S_ISDIR(sb.st_mode)) {
35  std::cout << output_prefix << " is not a directory " << std::endl;
36  } else {
37  validate_dir = true;
38  }
39  return validate_dir;
40 }
